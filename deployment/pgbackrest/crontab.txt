# pgBackRest backup schedule for mecontrola
# Full weekly backup (Sunday 05:00 UTC / 02:00 BRT)
0 5 * * 0 PGBACKREST_S3_KEY=${PGBACKREST_S3_KEY} PGBACKREST_S3_KEY_SECRET=${PGBACKREST_S3_KEY_SECRET} pgbackrest --stanza=mecontrola --type=full backup
# Differential daily backup (Monday-Saturday 05:00 UTC / 02:00 BRT)
0 5 * * 1-6 PGBACKREST_S3_KEY=${PGBACKREST_S3_KEY} PGBACKREST_S3_KEY_SECRET=${PGBACKREST_S3_KEY_SECRET} pgbackrest --stanza=mecontrola --type=diff backup
# Incremental backup every 6 hours
0 */6 * * * PGBACKREST_S3_KEY=${PGBACKREST_S3_KEY} PGBACKREST_S3_KEY_SECRET=${PGBACKREST_S3_KEY_SECRET} pgbackrest --stanza=mecontrola --type=incr backup
